Escrever um programa que aceite 2 inteiros a e b como inputs do usuário e retorne quais são os múltiplos comuns entre esses 2 inteiros.

Obs.: Um múltiplo comum é um número positivo inteiro, diferente de zero, que é múltiplo ao mesmo tempo entre 2 ou mais inteiros.

import java.util.Scanner;

public class Mainz {

    public static void main(String[] args) {

        Scanner scanner = new Scanner(System.in);

        System.out.println("Type number A:");
        int a = scanner.nextInt();

        System.out.println("Type number B:");
        int b = scanner.nextInt();

        scanner.close();
        System.out.printf("The common factors between %d and %d are ", a, b);

        for (int i = 1; i <= b; i++) {
            if (a % i == 0 && b % i == 0) {
                System.out.print(i + ", ");
            }
        }
    }
}
